--// Fungsi teleport ke posisi
local function teleportTo(pos)
    local char = LocalPlayer.Character
    if not char then return false end
    local hrp = char:FindFirstChild("HumanoidRootPart") or char.PrimaryPart
    if not hrp then return false end

    pcall(function()
        hrp.AssemblyLinearVelocity = Vector3.zero
        hrp.AssemblyAngularVelocity = Vector3.zero
        hrp.Velocity = Vector3.zero
    end)

    pcall(function()
        char:SetPrimaryPartCFrame(CFrame.new(pos))
    end)
    task.wait()
    return true
end

--// Start teleport sesuai mode
local function startTeleport()
    local map = {BC=1, CP1=2, CP2=3, CP3=4, CP4=5, CP5=6, CP6=7, CP7=8,}
    local startIndex = map[selectedCPMode] or 1

    repeat
        for i=startIndex,#teleportList do
            if not running then break end
            for t=teleportDelays[i],1,-1 do
                if not running then break end
              if statusUpdater then
    -- üî• Tambahan Nizar Loading Bar Block (tidak hapus sistem lain)
    local total = teleportDelays[i]
    local filled = math.floor(((total - t) / total) * 10)
    local bar = string.rep("‚ñà", filled) .. string.rep("‚ñë", 10 - filled)
    statusUpdater("Next teleport in: "..t.."s  ["..bar.."]")
end
                wait(1)
            end
            if not running then break end

            local pos = teleportList[i]
            if currentMode == "TELEPORT" then
                local success = teleportTo(pos)
                if statusUpdater then statusUpdater(success and "Teleported to "..i or "Teleport failed "..i) end
            else
                if LocalPlayer.Character and LocalPlayer.Character.PrimaryPart then
                    local char = LocalPlayer.Character
                    local oldCFrame = char.PrimaryPart.CFrame
                    local cam = workspace.CurrentCamera
                    local camCFrame = cam.CFrame
                    teleportTo(pos)
                    task.defer(function()
                        pcall(function()
                            char:SetPrimaryPartCFrame(oldCFrame)
                            cam.CFrame = camCFrame
                        end)
                    end)
                end
                if statusUpdater then statusUpdater("‚ö° GHOIB Quantum Done "..i) end
            end
        end
        wait(0.5)
    until not looping or not running

    if autoRespawn and LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("Humanoid") then
        LocalPlayer.Character.Humanoid.Health = 0
    end

    running = false
    if startBtn then
        startBtn.Text = "üöÄ START"
        startBtn.BackgroundColor3 = Color3.fromRGB(0,120,215)
    end
    if statusUpdater then statusUpdater("Status: Idle") end
end

                dragging=true; dragInput=input; mousePos=input.Position; framePos=mainFrame.Position

                    if input.UserInputState==Enum.UserInputState.End then dragging=false end
                end)
            end
        end)

            if input.UserInputType==Enum.UserInputType.MouseMovement or input.UserInputType==Enum.UserInputType.Touch then
                dragInput=input
            end

            if dragging and input==dragInput then
                local delta=input.Position-mousePos
                mainFrame.Position=UDim2.new(
                    framePos.X.Scale, framePos.X.Offset+delta.X,
                    framePos.Y.Scale, framePos.Y.Offset+delta.Y
